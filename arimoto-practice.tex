\section{Definitions and Problem Statement}
% ============================================================

\begin{frame}{Symbol Definitions}
    \begin{block}{Basic Symbols}
        \begin{center}
        \begin{tabular}{cl}
            \toprule
            \textbf{Symbol} & \textbf{Meaning} \\
            \midrule
            $X$ & Input random variable ($n$ discrete values) \\
            $Y$ & Output random variable ($m$ discrete values) \\
            $p(x_j)$ & Probability of input at $x_j$ \\
            $p(y_i)$ & Probability of output at $y_i$ \\
            $p(x_j, y_i)$ & Joint probability at $(x_j, y_i)$ \\
            $p(y_i \mid x_j)$ & Forward conditional probability (channel) \\
            $p(x_j \mid y_i)$ & Backward conditional probability (posterior) \\
            \bottomrule
        \end{tabular}
        \end{center}
    \end{block}
\end{frame}

\begin{frame}{Entropy and Mutual Information}
    \begin{block}{Entropy of Input}
        $$
        H(X) = \sum_{j=1}^{n} -p(x_j) \log\bigl(p(x_j)\bigr)
        $$
    \end{block}

    \begin{block}{Conditional Entropy $H(X \mid Y)$}
        $$
        H(X \mid Y) = \sum_{i=1}^{m} \sum_{j=1}^{n} -p(y_i \mid x_j)\, p(x_j)\, \log\!\left( \frac{p(y_i \mid x_j)\, p(x_j)}{\sum_{k=1}^{n} p(y_i \mid x_k)\, p(x_k)} \right)
        $$
    \end{block}

    \vspace*{-0.6em}
    \begin{block}{Mutual Information (non-symmetric: backward / posterior form)}
        \begin{align*}
            I(X; Y) &= H(X) - H(X \mid Y) \\
                    &= \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j)\, p(x_j) \log \frac{p(x_j \mid y_i)}{p(x_j)}
        \end{align*}
    \end{block}
\end{frame}

    \begin{frame}{Expansion of $H(X) - J(X \mid Y, \phi)$}
        \vspace*{-0.5em}
        \begin{block}{Definition of $J(X \mid Y, \phi)$}
            $$
            J(X \mid Y, \phi) = \sum_{i=1}^{m} \sum_{j=1}^{n} -p(y_i \mid x_j) p(x_j) \log \phi(x_j \mid y_i)
            $$
        \end{block}

        \vspace*{-1.4em}
        \small
        \setlength{\jot}{1pt}
        \begin{align*}
            H(X) - J(X \mid Y, \phi)
            &= \sum_{j=1}^{n} -p(x_j) \log p(x_j) \\
            &\quad - \left( \sum_{i=1}^{m} \sum_{j=1}^{n} -p(y_i \mid x_j) p(x_j) \log \phi(x_j \mid y_i) \right) \\
            &= \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j) p(x_j) \log \phi(x_j \mid y_i) \\
            &\quad - \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j) p(x_j) \log p(x_j) \\
            &= \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j) p(x_j) \log \frac{\phi(x_j \mid y_i)}{p(x_j)}
        \end{align*}
    \end{frame}

    \begin{frame}{Expansion of $H(X) - J(X \mid Y, \phi^t)$ at Iteration $t$}
        \vspace*{-0.5em}
        \begin{block}{Definition of $J(X \mid Y, \phi^t)$}
            $$
            J(X \mid Y, \phi^t) = \sum_{i=1}^{m} \sum_{j=1}^{n} -p(y_i \mid x_j) p^t(x_j) \log \phi^t(x_j \mid y_i)
            $$
        \end{block}

        \vspace*{-1.4em}
        \small
        \setlength{\jot}{1pt}
        \begin{align*}
            H(X) - J(X \mid Y, \phi^t)
            &= \sum_{j=1}^{n} -p^t(x_j) \log p^t(x_j) \\
            &\quad - \left( \sum_{i=1}^{m} \sum_{j=1}^{n} -p(y_i \mid x_j) p^t(x_j) \log \phi^t(x_j \mid y_i) \right) \\
            &= \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j) p^t(x_j) \log \phi^t(x_j \mid y_i) \\
            &\quad - \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j) p^t(x_j) \log p^t(x_j) \\
            &= \sum_{i=1}^{m} \sum_{j=1}^{n} p(y_i \mid x_j) p^t(x_j) \log \frac{\phi^t(x_j \mid y_i)}{p^t(x_j)}
        \end{align*}
    \end{frame}

\section{Iterative Algorithm of Computation}
% ============================================================

\begin{frame}{The Three-Step Iterative Algorithm}
    \begin{block}{Step 1: Initialization}
        Set the input distribution to be uniform:
        $$
            p^0(x_1) = p^0(x_2) = \cdots = p^0(x_n) = \frac{1}{n}
        $$
        Let $t = 0, 1, 2, \ldots$ denote the iteration index.
    \end{block}

    \begin{block}{Step 2: Update $\phi^t$ and Compute $C(t,t)$}
        Fix $p^t(x_j)$, find the optimal $\phi^t$:
        $$
            \phi^t(x_j \mid y_i) = \left[ \frac{p(y_i \mid x_j)\, p^t(x_j)}{\sum_{k=1}^{n} p(y_i \mid x_k)\, p^t(x_k)} \right]_{\forall\, i,j}
        $$
    \end{block}
\end{frame}

\begin{frame}{Step 2 (cont.): Compute Capacity $C(t,t)$}
    \begin{block}{Capacity at iteration $t$}
        \begin{align*}
            C(t,t) &= H(X) - H(X \mid Y) \\
                   &= \sum_{j=1}^{n} -p^t(x_j) \log p^t(x_j) \\
                   &\quad - \sum_{i=1}^{m}\sum_{j=1}^{n} -p(y_i \mid x_j)\, p^t(x_j) \log \frac{p(y_i\mid x_j)\, p^t(x_j)}{\sum_k p(y_i \mid x_k)\, p^t(x_k)}
        \end{align*}
    \end{block}

    \begin{exampleblock}{Interpretation}
        Since $\phi^t$ is already optimal at $p^t$, we get $J = H(X \mid Y)$. So $C(t,t)$ is the true mutual information at the current $p^t$.
    \end{exampleblock}
\end{frame}

\begin{frame}{Step 3: Update $p^{t+1}(x_j)$ via Lagrange}
    Now fix $\phi^t$ and maximize over $\{p(x_j)\}$ subject to $\sum_j p(x_j) = 1$.

    \begin{block}{Lagrangian Function}
        \begin{align*}
            F &= \sum_{j=1}^n -p^t(x_j) \log p^t(x_j) \\
              &\quad - \sum_{i,j} -p(y_i \mid x_j)\, p^t(x_j) \log \phi^t(x_j \mid y_i) \\
              &\quad + \lambda \Bigl( 1 - \sum_{j=1}^{n} p^t(x_j) \Bigr)
        \end{align*}
    \end{block}

    \begin{block}{Stationarity Conditions}
        $$
            \frac{\partial F}{\partial p^t(x_j)} = 0, \qquad \frac{\partial F}{\partial \lambda} = 0
        $$
    \end{block}
\end{frame}

\begin{frame}{Step 3: Taking Derivatives}
    Compute $\partial F / \partial p^t(x_j)$:
    \begin{align*}
        \frac{\partial F}{\partial p^t(x_j)} &= -\log p^t(x_j) - p^t(x_j) \cdot \frac{1}{p^t(x_j)} \\
        &\quad + \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) - \lambda \\
        &= -\log p^t(x_j) - 1 + \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) - \lambda
    \end{align*}

    \begin{exampleblock}{Setting $\partial F / \partial p^t(x_j) = 0$}
        $$
            \log p^{t+1}(x_j) = -1 - \lambda + \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i)
        $$
        $$
            p^{t+1}(x_j) = \exp\!\left( -1 - \lambda + \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right)
        $$
    \end{exampleblock}
\end{frame}

\begin{frame}{Step 3: Solving for $\lambda$}
    Use the constraint $\sum_{j=1}^{n} p^{t+1}(x_j) = 1$:
    \begin{align*}
        \sum_{j=1}^{n} \exp\!\left( -1 - \lambda + \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right) &= 1 \\
        \exp(-1 - \lambda) \cdot \sum_{j=1}^{n} \exp\!\left( \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right) &= 1
    \end{align*}
    $$
        \exp(-1 - \lambda) = \frac{1}{\sum_{j=1}^{n} \exp\!\left( \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right)}
    $$
\end{frame}

\begin{frame}[t]{Step 3: Final Update Rule}
    \footnotesize
    \begin{exampleblock}{Define}\vspace{-0.4em}
        $$
            r_j^t = \exp\!\left( \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right)
        $$
    \end{exampleblock}
    \vspace{-0.6em}
    \begin{block}{Update Formula for $p^{t+1}(x_j)$}\vspace{-0.4em}
        $$
            \boxed{\; p^{t+1}(x_j) = \frac{r_j^t}{\sum_{j=1}^{n} r_j^t} \;}
        $$
    \end{block}
    \vspace{-0.6em}
    \begin{block}{Updated Capacity (still uses $\phi^t$)}\vspace{-0.4em}
        $$
            \boxed{\; C(t+1, t) = \log \sum_{j=1}^{n} r_j^t \;}
        $$
        \vspace{-0.6em}

        Becomes $C(t+1,t+1)$ only after Step 2 of the next iteration recomputes $\phi^{t+1}$.
    \end{block}
\end{frame}

\begin{frame}{Derivation of $C(t+1, t) = \log \sum_j r_j^t$}
    \footnotesize
    \begin{align*}
        C(t+1, t) &= \sum_j -p^{t+1}(x_j) \log p^{t+1}(x_j) - \sum_{i,j} -p(y_i \mid x_j)\, p^{t+1}(x_j) \log \phi^t(x_j \mid y_i) \\
                  &= \sum_j -p^{t+1}(x_j) \log p^{t+1}(x_j) + \sum_j p^{t+1}(x_j) \log r_j^t \\
                  &= \sum_j p^{t+1}(x_j) \log \frac{r_j^t}{p^{t+1}(x_j)} \\
                  &= \sum_j \frac{r_j^t}{\sum_k r_k^t} \cdot \log \frac{r_j^t}{\,r_j^t \big/ \sum_k r_k^t\,} \\
                  &= \frac{\sum_j r_j^t}{\sum_k r_k^t} \cdot \log \sum_j r_j^t = \log \sum_{j=1}^{n} r_j^t
    \end{align*}
\end{frame}

\section{Example: 3$\times$3 Asymmetric Channel}
% ============================================================

\begin{frame}{Problem Setup: A $3 \times 3$ Channel}
    \begin{block}{Channel Transition Matrix $P(Y \mid X)$}
        $$
            P =
            \begin{bmatrix}
                p(y_1 \mid x_1) & p(y_1 \mid x_2) & p(y_1 \mid x_3) \\
                p(y_2 \mid x_1) & p(y_2 \mid x_2) & p(y_2 \mid x_3) \\
                p(y_3 \mid x_1) & p(y_3 \mid x_2) & p(y_3 \mid x_3)
            \end{bmatrix}
            =
            \begin{bmatrix}
                0.6 & 0.7 & 0.5 \\
                0.3 & 0.1 & 0.05 \\
                0.1 & 0.2 & 0.45
            \end{bmatrix}
        $$
    \end{block}

    \begin{alertblock}{This is NOT symmetric}
        Rows are not permutations of each other $\Rightarrow$ uniform input is \textbf{not} optimal.
    \end{alertblock}

    \begin{block}{Initialization}
        $$
            p^0(x_1) = p^0(x_2) = p^0(x_3) = \tfrac{1}{3}
        $$
    \end{block}
\end{frame}

\begin{frame}[t]{Iteration Loop Summary}
    \footnotesize
    \begin{exampleblock}{For each iteration $t = 0, 1, 2, \ldots$}
        \textbf{(1) Compute backward matrix $\phi^t$:}\vspace{-0.3em}
        $$
            \phi^t(x_j \mid y_i) = \frac{p(y_i \mid x_j)\, p^t(x_j)}{\sum_{k=1}^{3} p(y_i \mid x_k)\, p^t(x_k)}
        $$
        \vspace{-1.4em}

        \textbf{(2) Compute $r_j^t$:}\vspace{-0.3em}
        $$
            r_j^t = \exp\!\left( \sum_{i=1}^{3} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right)
        $$
        \vspace{-1.4em}

        \textbf{(3) Update input distribution:}\vspace{-0.3em}
        $$
            p^{t+1}(x_j) = \frac{r_j^t}{\sum_{j=1}^{3} r_j^t}
        $$
        \vspace{-1.4em}

        \textbf{(4) Capacity now (with the old $\phi^t$):}\vspace{-0.3em}
        $$
            C(t+1, t) = \log \sum_{j=1}^{3} r_j^t
        $$
        \vspace{-1.4em}

        \textbf{(5) Capacity after the next $\phi$-update $\phi^{t+1}$ (step (1) of iteration $t{+}1$):}\vspace{-0.3em}
        $$
            C(t+1, t+1) = \sum_{i,j} p(y_i \mid x_j)\, p^{t+1}(x_j) \log \frac{\phi^{t+1}(x_j \mid y_i)}{p^{t+1}(x_j)}
        $$
    \end{exampleblock}
\end{frame}

% ---------- Worked numerical iterations (t = 0, 1, 2) ----------
\begin{frame}{Iteration $t=0$: Compute $\phi^0$ and $C(0,0)$}
    \footnotesize
    \textit{All logarithms natural (units: nats).}\quad $p^0=\bigl(\tfrac13,\tfrac13,\tfrac13\bigr)$.
    \medskip

    \textbf{(1) Output marginals } $q(y_i)=\sum_k p(y_i\mid x_k)\,p^0(x_k)$:
    \begin{align*}
        q(y_1)&=(0.6+0.7+0.5)\tfrac13=0.6000, &
        q(y_2)&=(0.3+0.1+0.05)\tfrac13=0.1500,\\
        q(y_3)&=(0.1+0.2+0.45)\tfrac13=0.2500. &&
    \end{align*}

    \textbf{(2) Backward matrix } $\phi^0(x_j\mid y_i)=\dfrac{p(y_i\mid x_j)\,p^0(x_j)}{q(y_i)}$,\;
    e.g.\ $\phi^0(x_1\mid y_1)=\dfrac{0.6\cdot\frac13}{0.6}=0.3333$:
    $$
    \phi^0=
    \begin{bmatrix}
        0.3333 & 0.3889 & 0.2778 \\
        0.6667 & 0.2222 & 0.1111 \\
        0.1333 & 0.2667 & 0.6000
    \end{bmatrix}\ (\text{rows }y_i,\ \text{cols }x_j)
    $$

    \textbf{(3) Capacity at $p^0$:}
    $$
    C(0,0)=\sum_{i,j} p(y_i\mid x_j)\,p^0(x_j)\,\ln\frac{\phi^0(x_j\mid y_i)}{p^0(x_j)}=0.0858
    $$
\end{frame}

\begin{frame}{Iteration $t=0$: Update $p^1$ and $C(1,0)$}
    \footnotesize
    \textbf{(1) Compute } $r_j^0=\exp\!\Bigl(\sum_i p(y_i\mid x_j)\ln\phi^0(x_j\mid y_i)\Bigr)$,\\
    e.g.\ $r_1^0=\exp\bigl(0.6\ln0.3333+0.3\ln0.6667+0.1\ln0.1333\bigr)=\exp(-0.9823)=0.3745$:
    $$
    r^0=(0.3745,\ 0.3410,\ 0.3752),\qquad \textstyle\sum_j r_j^0=1.0907.
    $$

    \textbf{(2) Normalize } $p^1(x_j)=r_j^0\big/\sum_j r_j^0$:
    $$
    p^1=\Bigl(\tfrac{0.3745}{1.0907},\ \tfrac{0.3410}{1.0907},\ \tfrac{0.3752}{1.0907}\Bigr)=(0.3433,\ 0.3126,\ 0.3440).
    $$

    \textbf{(3) Updated capacity:}
    $$
    C(1,0)=\ln\textstyle\sum_j r_j^0=\ln 1.0907=0.0868\quad(>C(0,0)).
    $$
\end{frame}

\begin{frame}{Iteration $t=1$: Compute $\phi^1$ and $C(1,1)$}
    \footnotesize
    $p^1=(0.3433,\ 0.3126,\ 0.3440)$.
    \medskip

    \textbf{(1) Output marginals } $q(y_i)=\sum_k p(y_i\mid x_k)\,p^1(x_k)$:
    $$
    q(y_1)=0.5969,\quad q(y_2)=0.1515,\quad q(y_3)=0.2517.
    $$

    \textbf{(2) Backward matrix } $\phi^1(x_j\mid y_i)=\dfrac{p(y_i\mid x_j)\,p^1(x_j)}{q(y_i)}$,\;
    e.g.\ $\phi^1(x_1\mid y_1)=\dfrac{0.6\cdot0.3433}{0.5969}=0.3451$:
    $$
    \phi^1=
    \begin{bmatrix}

    \footnotesize
    $p^2=(0.3528,\ 0.2931,\ 0.3541)$.
    \medskip

    \medskip

    \alert{The pattern is now clear:}\ $p(x_2)$ keeps shrinking; $p(x_1),p(x_3)$ keep growing; $C$ rises at every half-step. The same two moves ($\phi$-update, $p$-update) just repeat until convergence.
\end{frame}

\begin{frame}{Stopping Criterion}
    \begin{block}{Convergence Test}
        Stop iterating when
        $$
            \bigl| C(t+1, t+1) - C(t, t) \bigr| < \varepsilon
        $$
        where $\varepsilon = 10^{-6}$ (project requirement).
    \end{block}

    \begin{alertblock}{Note}
        \begin{itemize}
            \item Convergence is guaranteed but can be slow.
            \item Each iteration is $O(nm)$ in operations.
            \item Far more efficient than direct optimization over the $(n-1)$-dim simplex.
        \end{itemize}
    \end{alertblock}
\end{frame}

\section{Summary}
% ============================================================

\begin{frame}{Summary of Arimoto's Algorithm}
    \begin{block}{Core Insight}
        Two-variable joint optimization $\Rightarrow$ alternating maximization
        \begin{itemize}
            \item Inner: $\phi^t(x_j \mid y_i) = \dfrac{p(y_i \mid x_j)\, p^t(x_j)}{\sum_k p(y_i \mid x_k)\, p^t(x_k)}$
            \item Outer: $p^{t+1}(x_j) = \dfrac{r_j^t}{\sum_j r_j^t}$
        \end{itemize}
    \end{block}

    \begin{block}{Key Properties}
        \begin{itemize}
            \item Monotonic increase: $C(t, t) \nearrow C(P)$
            \item Provably convergent to global optimum (convexity of mutual information in $p(x)$ for fixed $p(y \mid x)$)
            \item Simple to implement, only matrix-vector operations
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Recap of Important Formulas}
    \footnotesize
    \begin{block}{The 4 Essential Equations}
        $$
            \phi^t(x_j \mid y_i) = \frac{p(y_i \mid x_j)\, p^t(x_j)}{\sum_k p(y_i \mid x_k)\, p^t(x_k)}
        $$
        $$
            r_j^t = \exp\!\left( \sum_{i=1}^{m} p(y_i \mid x_j) \log \phi^t(x_j \mid y_i) \right)
        $$
        $$
            p^{t+1}(x_j) = \frac{r_j^t}{\sum_j r_j^t}
        $$
        $$
            C(t+1, t) = \log \sum_{j=1}^{n} r_j^t
        $$
    \end{block}
\end{frame}

\section*{Appendix: Decoupling via the Auxiliary Distribution}

\begin{frame}{Appendix: The Coupling Problem}
    \begin{block}{Expanding $I(X;Y) = H(X) - H(X \mid Y)$}
        Starting from
        $$
            I(X;Y) = \sum_{x,y} p(x)\,p(y\mid x) \log \frac{p(x\mid y)}{p(x)},
        $$
        substitute Bayes' rule $p(x\mid y) = \dfrac{p(x)\,p(y\mid x)}{\sum_{x'} p(x')\,p(y\mid x')}$:
        $$
            I(X;Y) = \sum_{x,y} p(x)\,p(y\mid x)\, \log \frac{p(y\mid x)}{\sum_{x'} p(x')\,p(y\mid x')}.
        $$
    \end{block}

    \begin{alertblock}{Source of Coupling}
        The denominator $\sum_{x'} p(x')\,p(y\mid x')$ \textbf{locks all input probabilities inside the same log}: differentiating w.r.t.\ any single $p(x_j)$ yields a nonlinear system — no closed-form solution by direct optimization.
    \end{alertblock}
\end{frame}

\begin{frame}{Appendix: Variational Lower Bound via $\phi(x \mid y)$}
    \begin{block}{Introduce an Auxiliary Distribution $\phi(x \mid y)$}
        Consider the gap between the true mutual information and a $\phi$-based surrogate:
        $$
            I(X;Y) - \sum_{x,y} p(x)\,p(y\mid x) \log \frac{\phi(x\mid y)}{p(x)} = \sum_{x,y} p(x)\,p(y\mid x) \log \frac{p(x\mid y)}{\phi(x\mid y)}
        $$
        The RHS is a \textbf{conditional KL divergence} $\geq 0$, equality iff $\phi(x\mid y) = p(x\mid y)$.
    \end{block}

    \begin{exampleblock}{Variational Form of Mutual Information}
        $$
            I(X;Y) = \max_{\phi}\; \sum_{x,y} p(x)\,p(y\mid x) \log \frac{\phi(x\mid y)}{p(x)}
        $$
    \end{exampleblock}
\end{frame}

\begin{frame}{Appendix: How Decoupling Happens}
    \begin{block}{Fixing $\phi(x\mid y)$, Maximizing over $p(x)$}
        With $\phi$ held fixed, the term $\dfrac{\phi(x\mid y)}{p(x)}$ depends on each $p(x_j)$ \textbf{independently}.
        \begin{itemize}
            \item The troublesome $\sum_{x'}$ inside the log is \textbf{gone}.
            \item The stationary condition $\partial / \partial p(x_j) = 0$ separates into one equation per $j$.
            \item $\Rightarrow$ closed-form update $p^{t+1}(x_j) = r_j^t / \sum_k r_k^t$.
        \end{itemize}
    \end{block}

    \begin{alertblock}{Geometric Interpretation (Lagrange)}
        At the optimum, the gradient of the objective is parallel to the gradient of the simplex constraint $\sum_j p(x_j)=1$. Alternating max over $(p, \phi)$ traces a path of monotonic improvement along this constrained manifold.
    \end{alertblock}
\end{frame}

\begin{frame}{Appendix: Proof of the Log-Sum Inequality}
    \begin{block}{Statement}
        For nonnegative reals $a_1,\ldots,a_n$ and $b_1,\ldots,b_n$ (with $b_i > 0$):
        $$
            \sum_{i=1}^{n} a_i \log \frac{a_i}{b_i} \;\geq\; \Bigl(\sum_{i=1}^{n} a_i\Bigr) \log \frac{\sum_{i=1}^{n} a_i}{\sum_{i=1}^{n} b_i}
        $$
        with equality iff $a_i / b_i$ is constant in $i$.
    \end{block}

    \begin{block}{Proof via Jensen's Inequality}
        Let $A = \sum_i a_i$, $B = \sum_i b_i$, and define weights $w_i = b_i / B$ (so $\sum_i w_i = 1$). Since $f(t) = t\log t$ is convex on $(0,\infty)$, by Jensen's inequality:
        $$
            \sum_i w_i\, f\!\left(\frac{a_i}{b_i}\right) \;\geq\; f\!\left(\sum_i w_i \cdot \frac{a_i}{b_i}\right) = f\!\left(\frac{A}{B}\right).
        $$
    \end{block}
\end{frame}

\begin{frame}{Appendix: Proof of the Log-Sum Inequality (cont.)}
    \begin{block}{Expand Both Sides}
        LHS: $\displaystyle \sum_i \frac{b_i}{B} \cdot \frac{a_i}{b_i} \log \frac{a_i}{b_i} = \frac{1}{B} \sum_i a_i \log \frac{a_i}{b_i}$.

        \vspace{0.3em}
        RHS: $\displaystyle \frac{A}{B} \log \frac{A}{B}$.

        \vspace{0.3em}
        Multiplying both sides by $B > 0$:
        $$
            \sum_i a_i \log \frac{a_i}{b_i} \;\geq\; A \log \frac{A}{B} = \Bigl(\sum_i a_i\Bigr) \log \frac{\sum_i a_i}{\sum_i b_i}. \qquad 
        $$
    \end{block}

    \begin{exampleblock}{Equality Condition}
        Jensen's inequality is tight iff the convex function is evaluated at a single point, i.e.\ $a_i / b_i$ is identical for all $i$ with $w_i>0$.
    \end{exampleblock}
\end{frame}
